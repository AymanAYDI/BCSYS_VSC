    PROCEDURE InsertRec@1(FileName@1001 : Text[250];VAR FileContent@1002 : InStream;PostExchDefCode@1003 : Code[20]);
    VAR
      PostingExchLineDef@1000 : Record 1227;
      OutStream@1005 : OutStream;
    BEGIN
      INIT;
      VALIDATE("File Name",FileName);
      "File Content".CREATEOUTSTREAM(OutStream);
      COPYSTREAM(OutStream,FileContent);
      VALIDATE("Posting Exch. Def Code",PostExchDefCode);
      PostingExchLineDef.SETRANGE("Posting Exch. Def Code",PostExchDefCode);
      IF PostingExchLineDef.FINDFIRST THEN

    END;

    PROCEDURE ImportFileContent@5(PostExchDef@1000 : Record 1222) : Boolean;
    VAR
      PostExchLineDef@1001 : Record 1227;
    BEGIN
      PostExchDef.TESTFIELD("Ext. Data Handling Codeunit");
      CODEUNIT.RUN(PostExchDef."Ext. Data Handling Codeunit",Rec);

      IF NOT "File Content".HASVALUE THEN
        EXIT(FALSE);
      PostExchLineDef.SETRANGE("Posting Exch. Def Code",PostExchDef.Code);
      PostExchLineDef.FINDFIRST;
      "Posting Exch. Def Code" := PostExchDef.Code;
      "Posting Exch. Line Def Code" := PostExchLineDef.Code;
      INSERT;
      EXIT(TRUE);
    END;

    PROCEDURE ImportToPostExch@2(PostExchDef@1000 : Record 1222) : Boolean;
    VAR
      Source@1001 : InStream;
      ProgressWindow@1002 : Dialog;
    BEGIN
      IF NOT "File Content".HASVALUE THEN
        IF NOT ImportFileContent(PostExchDef) THEN
          EXIT(FALSE);

      ProgressWindow.OPEN(ProgressWindowMsg);

      "File Content".CREATEINSTREAM(Source);
      SETRANGE("Entry No.","Entry No.");
      IF PostExchDef."Reading/Writing Codeunit" > 0 THEN
        CODEUNIT.RUN(PostExchDef."Reading/Writing Codeunit",Rec)
      ELSE BEGIN
        PostExchDef.TESTFIELD("Reading/Writing XMLport");
        XMLPORT.IMPORT(PostExchDef."Reading/Writing XMLport",Source,Rec);
      END;

      ProgressWindow.CLOSE;

      EXIT(TRUE);
    END;

    PROCEDURE ExportFromPostExch@3();
    VAR
      PostingExchDef@1001 : Record 1222;
      PostingExchMapping@1000 : Record 1224;
    BEGIN
      PostingExchMapping.SETRANGE("Posting Exch. Def Code","Posting Exch. Def Code");
      PostingExchMapping.SETRANGE("Table ID",DATABASE::"Payment Export Data");
      PostingExchMapping.FINDFIRST;

      PostingExchMapping.TESTFIELD("Mapping Codeunit");
      PostingExchDef.GET("Posting Exch. Def Code");
      PostingExchDef.TESTFIELD("Reading/Writing Codeunit");
      PostingExchDef.TESTFIELD("Ext. Data Handling Codeunit");

      IF PostingExchMapping."Pre-Mapping Codeunit" > 0 THEN
        CODEUNIT.RUN(PostingExchMapping."Pre-Mapping Codeunit",Rec);

      CODEUNIT.RUN(PostingExchMapping."Mapping Codeunit",Rec);

      IF PostingExchMapping."Post-Mapping Codeunit" > 0 THEN
        CODEUNIT.RUN(PostingExchMapping."Post-Mapping Codeunit",Rec);

      CODEUNIT.RUN(PostingExchDef."Reading/Writing Codeunit",Rec);

      CODEUNIT.RUN(PostingExchDef."Ext. Data Handling Codeunit",Rec);

      IF PostingExchDef."User Feedback Codeunit" > 0 THEN
        CODEUNIT.RUN(PostingExchDef."User Feedback Codeunit",Rec);
    END;

    BEGIN
    END.
